üß† Functional Quick Sort in Detail

This walkthrough demonstrates how to implement the classic quick sort algorithm without mutating the source data by composing FP-ASM primitives. We rely on streaming operations such as fp_partition_gt_i64 and fp_concat_i64 to express the recursive structure in a purely functional style.

  1. Pick a pivot without touching the original array (we simply read input[0]).
  2. Partition the tail into elements greater than the pivot and those less-than-or-equal using fp_partition_gt_i64.
  3. Isolate duplicates of the pivot with a second partition call, preserving the mathematical stability even when the pivot is INT64_MIN.
  4. Recursively sort the smaller sublists by returning freshly allocated arrays from each call.
  5. Concatenate the results via fp_concat_i64 so that the final array is built from immutable pieces.
#include <inttypes.h>
#include <stdint.h>
#include <stdio.h>
#include <stdlib.h>
#include "fp_core.h"

typedef struct {
    int64_t* data;
    size_t length;
} FPArray;

static FPArray fp_quicksort_pure(const int64_t* input, size_t n) {
    if (n == 0) {
        return (FPArray){ .data = NULL, .length = 0 };
    }
    if (n == 1) {
        int64_t* single = malloc(sizeof(int64_t));
        single[0] = input[0];
        return (FPArray){ .data = single, .length = 1 };
    }

    const int64_t pivot = input[0];

    int64_t* greater = malloc(n * sizeof(int64_t));
    int64_t* less_equal = malloc(n * sizeof(int64_t));
    size_t greater_count = 0;
    size_t less_equal_count = 0;

    fp_partition_gt_i64(input + 1, greater, less_equal, n - 1, pivot,
                        &greater_count, &less_equal_count);

    int64_t* equals = malloc(n * sizeof(int64_t));
    int64_t* less = malloc(n * sizeof(int64_t));
    size_t equal_count = 1; // include the pivot itself
    size_t less_count = 0;
    equals[0] = pivot;

    if (less_equal_count > 0) {
        if (pivot == INT64_MIN) {
            for (size_t i = 0; i < less_equal_count; ++i) {
                equals[equal_count++] = less_equal[i];
            }
        } else {
            size_t duplicate_count = 0;
            fp_partition_gt_i64(less_equal, equals + 1, less, less_equal_count, pivot - 1,
                                &duplicate_count, &less_count);
            equal_count += duplicate_count;
        }
    }
    free(less_equal);

    FPArray left = fp_quicksort_pure(less, less_count);
    FPArray right = fp_quicksort_pure(greater, greater_count);

    int64_t* left_with_pivot = malloc((left.length + equal_count) * sizeof(int64_t));
    size_t left_with_pivot_len = fp_concat_i64(left.data, equals, left_with_pivot,
                                               left.length, equal_count);

    int64_t* result = malloc((left_with_pivot_len + right.length) * sizeof(int64_t));
    size_t total_len = fp_concat_i64(left_with_pivot, right.data, result,
                                     left_with_pivot_len, right.length);

    free(left.data);
    free(right.data);
    free(left_with_pivot);
    free(greater);
    free(equals);
    free(less);

    return (FPArray){ .data = result, .length = total_len };
}

int main(void) {
    const int64_t raw[] = { 42, -7, 13, 13, 0, 5, INT64_MIN, 9, 42 };
    const size_t count = sizeof(raw) / sizeof(raw[0]);

    FPArray sorted = fp_quicksort_pure(raw, count);

    printf("Original: ");
    for (size_t i = 0; i < count; ++i) {
        printf("%" PRId64 "%s", raw[i], (i + 1 == count) ? "\n" : ", ");
    }

    printf("Sorted:   ");
    for (size_t i = 0; i < sorted.length; ++i) {
        printf("%" PRId64 "%s", sorted.data[i], (i + 1 == sorted.length) ? "\n" : ", ");
    }

    free(sorted.data);
    return 0;
}

Key insight: every recursive call returns a brand-new array, and we only use library helpers to stitch those immutable fragments together. The original raw data remains untouched throughout the entire run.
